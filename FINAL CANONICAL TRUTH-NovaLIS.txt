#!/usr/bin/env python3
"""
Advisory Terminology Checker

Scans governance-critical paths for agentic terminology.
Reports violations but does not fail CI (advisory only).
"""

import re
from pathlib import Path

# Governance-critical paths only
GOVERNANCE_PATHS = [
    "nova_core/memory/",
    "nova_core/governor/",
    ".governance/",
    "MEMORY_CONSTITUTIONAL_STRESS_TEST.md"
]

# Forbidden → Required (from stress test)
TERMINOLOGY_MAP = {
    "project": "Named Collection",
    "trigger": "Explicit Check",
    "reminder": "Scheduled Note",
    "status": "Stored Field",
    "searchable": "Explicit Lookup",
    "background": "User-Initiated Poll",
    "smart": "Deterministic",
    "learn": "Store"
}

def check_file(file_path: Path) -> list:
    """Check a single file for agentic terminology."""
    violations = []
    
    try:
        content = file_path.read_text()
        lines = content.split('\n')
        
        for i, line in enumerate(lines, 1):
            for forbidden, replacement in TERMINOLOGY_MAP.items():
                if re.search(rf'\b{forbidden}\b', line, re.IGNORECASE):
                    violations.append({
                        'file': str(file_path),
                        'line': i,
                        'term': forbidden,
                        'replacement': replacement,
                        'context': line.strip()[:80]
                    })
    
    except UnicodeDecodeError:
        pass
    
    return violations

def main():
    """Main advisory check."""
    print("🔍 Advisory Terminology Check (Governance Paths Only)")
    print("=" * 60)
    
    all_violations = []
    
    for path_pattern in GOVERNANCE_PATHS:
        for file_path in Path.cwd().glob(path_pattern):
            if file_path.is_file():
                violations = check_file(file_path)
                all_violations.extend(violations)
    
    if all_violations:
        print(f"⚠️  Found {len(all_violations)} potential terminology issues:")
        for v in all_violations:
            print(f"  {v['file']}:{v['line']} - '{v['term']}' → '{v['replacement']}'")
            print(f"     Context: {v['context']}")
        
        print("\n💡 Advisory: These terms may indicate agentic framing.")
        print("   Consider using constitutionally-mandated neutral terms.")
        return 0  # Advisory only - don't fail CI
    
    else:
        print("✅ No terminology issues found in governance paths.")
        return 0

if __name__ == "__main__":
    main()

"""
Behavior-Based Autonomy Vector Tests

Tests derived from MEMORY_CONSTITUTIONAL_STRESS_TEST.md
Focuses on actual behavior, not just vocabulary.
"""

import ast
import re
from pathlib import Path

class TestAutonomyVectors:
    """Test the 7 autonomy vectors are blocked."""
    
    def test_vector_1_no_temporal_tracking(self):
        """Vector 1: No time-based pattern tracking"""
        # Check that no code tracks when things happen or calculates frequencies
        memory_dir = Path("nova_core/memory/")
        
        if not memory_dir.exists():
            return  # No memory implementation yet - test passes
        
        for file in memory_dir.rglob("*.py"):
            content = file.read_text()
            
            # Check for time-based aggregation or frequency counting
            dangerous_patterns = [
                r"frequency",
                r"count.*over.*time",
                r"pattern.*time",
                r"usual.*time",
                r"average.*interval",
                r"historically",
                r"typically"
            ]
            
            for pattern in dangerous_patterns:
                assert not re.search(pattern, content, re.IGNORECASE), \
                    f"Found temporal tracking in {file}: {pattern}"
    
    def test_vector_3_no_usage_counters(self):
        """Vector 3: No access frequency tracking"""
        memory_dir = Path("nova_core/memory/")
        
        if not memory_dir.exists():
            return
        
        for file in memory_dir.rglob("*.py"):
            content = file.read_text()
            
            # Parse AST to look for access counters
            try:
                tree = ast.parse(content)
                
                for node in ast.walk(tree):
                    # Check for variable names indicating usage tracking
                    if isinstance(node, ast.Name):
                        var_name = node.id.lower()
                        if any(term in var_name for term in [
                            'access_count', 'usage_count', 'hit_count',
                            'popularity', 'frequency', 'priority_score'
                        ]):
                            raise AssertionError(
                                f"Found usage tracking variable in {file}: {var_name}"
                            )
                    
                    # Check for increment operations on counters
                    if isinstance(node, ast.AugAssign):
                        if isinstance(node.op, ast.Add):
                            # Looking for access_count += 1 patterns
                            if isinstance(node.target, ast.Name):
                                if 'count' in node.target.id.lower():
                                    raise AssertionError(
                                        f"Found counter increment in {file}: {node.target.id}"
                                    )
                
            except SyntaxError:
                continue
    
    def test_vector_2_no_relationship_inference(self):
        """Vector 2: No automatic relationship mapping"""
        memory_dir = Path("nova_core/memory/")
        
        if not memory_dir.exists():
            return
        
        for file in memory_dir.rglob("*.py"):
            content = file.read_text()
            
            # Check for functions that connect items
            dangerous_patterns = [
                r"connect.*items",
                r"relate.*items",
                r"link.*automatically",
                r"infer.*relationship",
                r"find.*connections",
                r"correlate.*items"
            ]
            
            for pattern in dangerous_patterns:
                assert not re.search(pattern, content, re.IGNORECASE), \
                    f"Found relationship inference in {file}: {pattern}"
    
    def test_vector_5_no_background_processes(self):
        """Vector 5: No background checks or polling"""
        # Check that memory system doesn't start threads or timers
        memory_dir = Path("nova_core/memory/")
        
        if not memory_dir.exists():
            return
        
        for file in memory_dir.rglob("*.py"):
            content = file.read_text()
            
            # Look for background process patterns
            dangerous_imports = [
                "threading", "asyncio", "schedule", "apscheduler",
                "Timer", "thread", "loop", "background"
            ]
            
            for imp in dangerous_imports:
                if f"import {imp}" in content or f"from {imp}" in content:
                    # Check if it's used for memory operations
                    tree = ast.parse(content)
                    for node in ast.walk(tree):
                        if isinstance(node, ast.Call):
                            # Check if threading/asyncio is called
                            if isinstance(node.func, ast.Name):
                                if node.func.id in ['Thread', 'Timer', 'start']:
                                    raise AssertionError(
                                        f"Found background process in memory code: {file}"
                                    )
