#!/usr/bin/env python3
"""
Verifies that break test harness respects constitutional constraints.
Run this BEFORE running any break attempts.
"""

import ast
import inspect
import os

class ConstitutionalTestVerifier:
    """Ensures tests don't cheat."""
    
    FORBIDDEN_PATTERNS = [
        # Direct internal calls
        "Governor._internal_process",
        "Governor._entry_point_verification",
        "governor.constitution.NormativeLanguageBan",  # Should only check output, not call directly
        
        # Privilege escalation
        "os.setuid", "os.seteuid",
        "import sys; sys.modules",
        "__import__",
        
        # Phase manipulation
        "set_phase", "change_phase", "override_phase",
        
        # Audit bypass
        "skip_audit", "no_log", "silent_mode"
    ]
    
    @staticmethod
    def verify_test_file(filepath: str) -> bool:
        """Returns True if test file is constitutionally valid."""
        with open(filepath, 'r') as f:
            content = f.read()
        
        # Parse AST to ensure no cheating
        tree = ast.parse(content)
        
        # Check for direct Governor internals usage
        for node in ast.walk(tree):
            if isinstance(node, ast.Call):
                # Check function calls
                if isinstance(node.func, ast.Attribute):
                    full_name = ast.unparse(node.func)
                    for forbidden in ConstitutionalTestVerifier.FORBIDDEN_PATTERNS:
                        if forbidden in full_name:
                            print(f"❌ VIOLATION in {filepath}: Direct call to {full_name}")
                            return False
        
        # Check imports
        for node in ast.walk(tree):
            if isinstance(node, ast.Import):
                for name in node.names:
                    if "governor._internal" in name.name:
                        print(f"❌ VIOLATION in {filepath}: Import of internal module {name.name}")
                        return False
        
        return True

                        print(f"\n❌ FAILED: {filepath} is constitutionally invalid.")
                        print("Fix before running break attempts.")
                        return False
        
        print("✅ All break attempts are constitutionally valid.")
        return True
